    # 104二叉树的最大深度

    # 题目

    给定一个二叉树，找出其最大深度.

    二叉树的深度为根节点到最远叶子节点的最长路径上的节点数。

    # 用例

    [3, 9, 20, null, null, 15, 7];

    3

    /
    9 20 /
    15 7

    返回最大深度3

    # 解题思路

    • 求最大深度，考虑使用深度优先遍历
    • 遍历过程中，记录每个节点所在的层级，找出最大的层级。

    # 解题步骤

    • 新建一个变量，记录最大的深度。
    • 在遍历过程中，不断的刷新最大深度这个变量。
    • 返回变量。

    # coding

    **
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function(root) {
    // 记录最大深度
    let res = 0;
    // 深度优先遍历
    const dfs = (n, l) => {
        if (!n) return;
        
        // 记录每一个节点的层级
        // res = Math.max(res, l);

        // 优化，只需要对叶子节点来进行比较大值
        if (!n.left && !n.right) {
            res = Math.max(res, l);
        }
        dfs(n.left, l + 1);
        dfs(n.right, l + 1);
    }
    // 调用，并传入层级1
    dfs(root, 1); 
    // 调用，传入root
    // 返回最大深度遍历
}
