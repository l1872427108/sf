    # 102二叉树的层序遍历

    # 题目

    给你二叉树的根节点 root，返回其节点值的层序遍历。（逐层地，从左到右访问所有节点）。

    # 解题思路

    • 逐层，就是使用广度优先遍历
    • 记录当前节点的层级。

    # 解题步骤

    # 用例

    输入：root = [3,9,20,null,null,15,7] 输出：[[3],[9,20],[15,7]]

    3

    /
    9 20 /
    15 7

    # coding

    var levelOrder = function (root) {
    if (!root) return [];
    const q = [[root, 0]];
    const res = [];
    while (q.length) {
        const [n, level] = q.shift();
        if (!res[level]) {
            res.push([n.val]);
        } else {
            res[level].push(n.val);
        }
        if (n.left) q.push([n.left, level]);
        if (n.right) q.push([n.right, level]);
    }
    return res;
}
